CMD Line

git init

	Initialized empty git repository in directory

git status

	No commits yet

git add file.txt
git status

	No commits yet
	Changes to be commited file.txt (In staging area)

git commit -m "message"


git diff

	View changes saved in working directory from last git state

git checkout file.txt 

	Rollback changes to last git commit

head hash = current hash
